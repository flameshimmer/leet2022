	    int minSubArrayLen(int target, vector<int>& nums) {
			int len = nums.size();
			if (len == 0) { return 0; }

			int result = INT_MAX;
			int start = 0;
			int sum = 0;
			for (int end = 0; end < len; end++) {
				sum += nums[end];
				while (start <= end && sum >= target) {
					result = min(result, end - start + 1);
					sum -= nums[start];
					start++;
				}
			}
			return result == INT_MAX ? 0 : result;
	    }
